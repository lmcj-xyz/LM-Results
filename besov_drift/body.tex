\begin{definition}{Local time at zero}\label{def:local_time_zero}
  For any real-valued continuous semi-martingale
	$ Z $,
	the local time at zero
  $L^{0}_{t} (\bar{Y})$
  is defined as
  \begin{equation}
    L^0_t (Z) = \lim_{\epsilon \to 0} \frac{1}{2 \epsilon} \int_0^t \mathbbm{1}_{\{| Z | \leq \epsilon\}} d \langle Z \rangle_s
    \text{, }
    \mathbb{P}
    \text{-a.s.}
  \end{equation}
  For all
  $t \geq 0$.
\end{definition}

\begin{definition}{Kolmogorov equations}\label{def:kolmogorov_eqns}
  For
  $\beta \in (0, 1/2)$
  let
  $b \in C_T \mathcal{C}^{-\beta}$,
  $u, u^N \in C_T\mathcal{C}^{(1 + \beta)+}$, 
  and
  $b^N \to b$
  as
  $N \to \infty$
  in
  $C_T \mathcal{C}^{-\beta}$
	The equations

	  \label{eq:kolmogorov}
	  \begin{cases}
	    \partial_t u_i + \frac{1}{2} \Delta u_i + b_i \nabla u_i = \lambda u_i - b_i
	    \\
	    u_i(T) = 0,

\section{Bounds for the difference of solutions to the Kolmogorov equations}

We need a bound for
$u - u^N$
and
$\nabla u - \nabla u^N$
in
$L_\infty$ 
for the case in which 
$u \in C_T \mathcal{C}^{1+\alpha}$
for some
$\alpha > \beta$
which is an adaptation of
\cite[Lemma 5.2]{de_angelis_numerical_2020}.

The result builds on top of the following result:

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Bound for the rho-equivalent norm of the difference u-uN
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{proposition}
{Bound for the $\rho$-equivalent norm of $u - u^N$}
\label{prop:diff_u_uN}

Let
$u, u^N$
be (mild) solutions to the Kolmogorov equations from
Definition \ref{def:kolmogorov_eqns}
then as
$N \to \infty$

\begin{equation}
\| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}^{1 + \alpha}}
	\leq
	\frac{c T^{\frac{1 - \beta - \alpha}{2}} \| b_i - b^N_i \|_{C_T\mathcal{C}^{-\beta}} (\| u_i \|_{C_T\mathcal{C}^{1 + \alpha}} - 1) }{1 - c \rho^{\frac{\alpha + \beta - 1}{2}} (\| b \|_{C_T\mathcal{C}^{-\beta}} + \lambda)}
\end{equation}

for
$\rho \geq \rho_0$,
where

\begin{equation}
	\rho_0 = 2 c ( \|  b_i \|_{C_T \mathcal{1+\alpha}} + \lambda ) ^ { \frac{ 2 }{ \alpha + \beta + 1}}
\end{equation}

and
$\lambda > 0$.

\end{proposition}

\begin{proof}

	See that
	$u^N(T) = u(T) = 0$,
	and in
	\cite{issoglio_pde_nodate},
	set
	$g^N, g$
	as
	$b^N, b$
	respectively.
	See that
	$b^N \to b$.
	Then let us reformulate the rest of the aforementioned result for
	$\lambda \neq 0$.

	As
	$u^N, u$
	are mild solutions, we have

	\begin{align*}
		u_i(t) - u^N_i(t)
		&=
		P_{T-t} (u_i(T) - u^N_i(T))
		\\
		&+
		\int_t^T P_{s-t}(\nabla u_i b_i - \nabla u^N_i b^N_i) ds
		\\
		&-
		\int_t^T P_{s-t}(\lambda u_i + b_i - \lambda u^N_i + b^N_i) ds
		\\
		&=
		\int_t^T P_{s-t}(\nabla u_i b_i - \nabla u^N_i b^N_i) ds
		\\
		&-
		\lambda  \int_t^T P_{s-t}(u_i - u^N_i) ds
		\\
		&-
		\int_t^T P_{s-t}(b_i - b^N_i) ds
		\\
		&=
		\int_t^T P_{s-t}[(\nabla u_i b_i - \nabla u_i b^N_i) + (\nabla u_i b^N_i - \nabla u^N_i b^N_i)] ds
		\\
		&-
		\lambda  \int_t^T P_{s-t}(u_i - u^N_i) ds
		\\
		&-
		\int_t^T P_{s-t}(b_i - b^N_i) ds
		\\
		&=
		\int_t^T P_{s-t}(\nabla u_i b_i - \nabla u_i b^N_i)  ds
		\\
		&+
		\int_t^T P_{s-t}(\nabla u_i b^N_i - \nabla u^N_i b^N_i) ds
		\\
		&-
		\lambda  \int_t^T P_{s-t}(u_i - u^N_i) ds
		\\
		&-
		\int_t^T P_{s-t}(b_i - b^N_i) ds
	\end{align*}

	Now let us compute the $\rho$-equivalent norm of
	$u-u^N$,
	for some
	$\alpha > \beta$
	
	
    %\todo{this norm is wrong, should be $1+\alpha$ on the lhs and everywhere else}
    {\color{red}!!!!!!!!!!!!!!!!!!!!!}\lm{this norm is wrong, should be $1+\alpha$ on the lhs and everywhere else}    {\color{red}!!!!!!!!!!!!!!!!!!!!!}
	\begin{align*}
		\| u_i - u^N_i \| ^ {(\rho)}_{C_T \mathcal{C}^{-\beta}}
		&=
		\sup_{0\leq t \leq T} e^{-\rho(T-t)} \| u(t) - u^N(t) \|_{C_T \mathcal{C}^{1 + \alpha}}
		\\
		&\leq
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} \| \int_t^T P_{s-t}(\nabla u_i b_i - \nabla u_i b^N_i)  ds \|_{C_T \mathcal{C}^{1 + \alpha}}
	     	\\ 
		&+
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} \| \int_t^T P_{s-t}(\nabla u_i b^N_i - \nabla u^N_i b^N_i) ds\|_{C_T \mathcal{C}^{1 + \alpha}}
	     	\\ 
		&-
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} \| \lambda  \int_t^T P_{s-t}(u_i - u^N_i) ds\|_{C_T \mathcal{C}^{1 + \alpha}}
	     	\\ 
		&-
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} \| \int_t^T P_{s-t}(b_i - b^N_i) ds\|_{C_T \mathcal{C}^{1 + \alpha}}
		.
	\end{align*}

	Let us take each term from the right hand side of the inequality and bound them.

	For the first term, using
	$\gamma + 2\theta = 1 + \alpha$,
	$\gamma = -\beta$,
	$\theta = \frac{1 + \alpha + \beta}{2}$,
	$\| P_t f \|_{\gamma + 2 \theta} \leq c t ^{-\theta} \| f \|_{\gamma} \|$
	and
	$\| \nabla g \|_\xi \leq c \|g \|_{\xi + 1}$

	\begin{align*}
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} 
		&\left\| \int_t^T P_{s-t} \left(\nabla u_i b_i - \nabla u_i b^N_i \right)  ds \right\|_{C_T \mathcal{C}^{1+\alpha}}
		\\
		&\leq
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} \int_t^T (s - t)^{-\theta} \| \nabla u_i\|_{C_T \mathcal{C}^\alpha} \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}}
		\\
		&\leq
		c \| u_i \|_{C_T \mathcal{C}_{1 + \alpha}} \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}} \sup_{0 \leq t \leq T} e^{-\rho(T-t)}(T-t)^{\frac{1 - \beta -\alpha}{2}}
		\\
		&\leq
		c T^{\frac{1 - \beta - \alpha}{2}} \| u_i \|_{C_T \mathcal{C}^{1 + \alpha}} \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}}
	\end{align*}

	For the second term, see that for
	$N \to \infty$,
	we have
	$\| b^N \|_{C_T \mathcal{C}^{-\beta}} \leq 2 \| b \|_{C_T \mathcal{C}^{-\beta}}$

	\begin{align*}
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} 
		&\left\| \int_t^T P_{s-t} b^N_i(\nabla u_i - \nabla u^N_i ) ds\right\|_{C_T \mathcal{C}^{1+\alpha}}
		\\
		&\leq
		c \sup_{0\leq t\leq T} \int_t^T(s - t)^{-\theta} e^{-\rho(T-t)} 2 \| b_i \|_{C_T \mathcal{C}^{-\beta}} \| \nabla u_i - \nabla u^N_i \|_{C_T \mathcal{C}^{1+\alpha}} ds
		\\
		&\leq
		c \| b_i \|_{C_T \mathcal{C}^{-\beta}} \| u_i - u^N_i \|_{C_T \mathcal{C}^{-\beta}}^{(\rho)} \int_t^T(s - t)^{-\theta} e^{-\rho(T-t)} ds
		\\
		&\leq
		c \| b_i \|_{C_T \mathcal{C}^{-\beta}} \| u_i - u^N_i \|_{C_T \mathcal{C}^{-\beta}}^{(\rho)} \rho^{\frac{\alpha + \beta - 1}{2}}
	\end{align*}

	For the third term, which is the one that difers from the proof in
	\cite{issoglio_pde_nodate}

	$\| P_t f \|_{\gamma + 2\theta} \leq ct^{-\theta} \| f \|_{\gamma}$,

	$\gamma + 2 \theta = 1 + \alpha$
	and
	$\gamma = 1 + \alpha$,
	so that
	$\theta = 0$
	because
	$u, u^N \in C_T \mathcal{C}^{1+\alpha}$,
	so we will have

	\begin{align*}
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} 
		&\left\| \lambda  \int_t^T P_{s-t}(u_i - u^N_i) ds\right\|_{1+\alpha}
		\\
		&\leq
		c \lambda \sup_{0\leq t\leq T} e ^{-\rho(T-t)} \int_t^T (s - t)^{-0} \| u_i - u^N_i \|_{1 + \alpha} ds
		\\
		&=
		c \lambda \sup_{0\leq t\leq T} e ^{-\rho(T-t)} \int_t^T e ^{-\rho(T-s)} \sup_{0\leq s\leq T} e ^{-\rho(T-s)} \| u_i - u^N_i \|_{1 + \alpha} ds
		\\
		&=
	c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \int_t^T e ^{-\rho(T-s)} e ^{-\rho(T-t)} ds
		\\
		&=
	c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \int_t^T e^{-\rho(s-t)} ds
		\\
		&=
		c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \sup_{0\leq t\leq T} \rho^{-1} [1 - e^{-\rho(T-t)}]
		\\
		&\leq
		c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \rho^{-1}
		\\
		&\leq
		c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \rho^{\frac{\alpha+\beta-1}{2}}
	\end{align*}

	And for the last term
	\begin{align*}
		\sup_{0\leq t\leq T} e ^{-\rho(T-t)} &\left\| \int_t^T P_{T-s}(b_i - b^N_i) ds\right\|_{C_T \mathcal{C}^{1+\alpha}}
		\\
		&\leq
		c \sup_{0\leq t\leq T} e ^{-\rho(T-t)} \int_t^T (s - t)^{-\frac{\alpha+\beta-1}{2}} \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}} ds
		\\
		&\leq
		c \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}} \sup_{0\leq t\leq T} e ^{-\rho(T-t)} (s - t)^{-\frac{\alpha+\beta-1}{2}}
		\\
		&\leq
		c T^{\frac{1-\beta-\alpha}{2}} \| b_i - b^N_i \|_{C^T \mathcal{C}^{-\beta}}
	\end{align*}

	Putting everything together

	\begin{align*}
		\| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}^{-\beta}} 
		&\leq
		c T^{\frac{1-\beta-\alpha}{2}} \|u_i\|_{C_T \mathcal{C}^{1+\alpha}} \| b_i - b^N_i \|_{C_T \mathcal{C}^{-\beta}}
			\\
		&+
		c \| b_i \|_{C_T \mathcal{C}^{-\beta}} \| u_i - u^N_i \|_{C_T \mathcal{C}^{-\beta}}^{(\rho)} \rho^{\frac{\alpha + \beta - 1}{2}}
		\\
		&-
		c \lambda \sup_{0\leq t\leq T} \| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}_{1 + \alpha}} \rho^{\frac{\alpha+\beta-1}{2}}
		\\
		&-
		c T^{\frac{1-\beta-\alpha}{2}} \| b_i - b^N_i \|_{C^T \mathcal{C}^{-\beta}},
		\\
	\end{align*}

		\\
		&\leq
		cT^{\frac{1-\beta-\alpha}{2}} \| b_i - b^N_i \|_{C^T \mathcal{C}^{-\beta}} (\| u_i \|_{C_T \mathcal{C}^{1+\alpha}} -1)
		\\
		\| u_i - u^N_i \|^{(\rho)}_{C_T \mathcal{C}^{-\beta}}

	\end{align*}

	As required.
\end{proof}

\begin{proposition}
{Bounds for $\|u - u^N\|_{L_\infty}$ and $\|\nabla u - \nabla u^N\|_{L_\infty}$}
\label{prop:diff_uN_graduN}

Let
$\beta \in (0, 1/2)$
and
$b \in C_T \mathcal{C}^{-\beta}$.
Let 
$u, u^N \in C_T\mathcal{C}^{(1 + \beta)+}$
be (mild) solutions to the Kolmogorov equations from
Definition \ref{def:kolmogorov_eqns}.

Assume, by
Proposition \ref{prop:diff_u_uN},
that for some 
$\alpha > \beta$

\begin{equation}
	\| u - u^N \|^{(\rho)}_{C_{T} \mathcal{C}^{1+\alpha}}
	\leq
	c(\rho)
	\| b - b^N \|_{C_{T} \mathcal{C}^{-\beta}}
	.
	\label{eq:u-uNb-bN}
\end{equation}

With
$c(\rho)$
as in
Proposition \ref{prop:diff_u_uN}
and
$\rho_0$
is large enough such that
$c(\rho) > 0$
for all
$\rho > \rho_0$.

Then for all
$t \in [0, T]$

\begin{equation}
	\| u^N(t) - u(t) \|_{L^\infty}
	\leq
	\kappa_\rho \| b - b^N \|_{C_T\mathcal{C}^{-\beta}}
	\label{eq:uNu_bounded_by_bNb}
\end{equation}

\begin{equation}
	\| \nabla u^N(t) - \nabla u(t) \|_{L^\infty}
	\leq
	\kappa_\rho \| b - b^N \|_{C_T\mathcal{C}^{-\beta}}
	\label{eq:graduNu_bounded_by_bNb}
\end{equation}
with
$\kappa_\rho = c \cdot c(\rho) \cdot e^{\rho T}$.

\end{proposition}

\begin{proof}
    First let us prove
    \eqref{eq:uNu_bounded_by_bNb}.
    
    Let
    $t \in [0, T]$,
    and see that since
    $u, u^N \in C_T \mathcal{C}^{(1 + \beta)+}$
    there exists
    $\alpha > \beta$
    such that
    $u, u^N \in C_T \mathcal{C}^{1 + \alpha}$,
    then for any
    $f \in \mathcal{C}^{1 + \alpha}$
    we have
    \begin{equation}
        \| f \|_{C^{1 + \alpha}}
        \leq
        c
        \left(
        \sup_{x \in \mathbb{R}^d} | f(x) |
        + \sup_{x \neq y \in \mathbb{R}^d}
        \frac{|\nabla f (x) - \nabla f(y)|}{|x - y|^\alpha}
        \right)
    \end{equation}
    
    so we have
    
    \begin{equation}
    \begin{split}
        \| u^N(t) - u(t) \|_{L^\infty}
        &=
        \sup_{x \in \mathbb{R}^d} | u^N(t, x) - u(t, x) |
        \\
        &\leq
        c \| u^N(t) - u(t) \|_{\mathcal{C}^{\alpha + 1}}
    \end{split}
    \label{eq:u-uN_in_Linfinity_bounded_by_u-uN_in_Calpha1}
    \end{equation}
    
    Moreover, using the
    $(\rho)$-equivalent
    norm
    \begin{equation}
        \| f \|_{\mathcal{C}^{1 + \alpha}} = \sup_{t \in [0, T]} e^{- \rho (T - t)} \left\|f(t)\right\|_{\mathcal{C}^{1 + \alpha}},
    \end{equation}
    and
    \eqref{eq:u-uNb-bN}
    we see that
    \begin{equation}
        \begin{split}
            \left\|u^N - u\right\|_{C_T \mathcal{C}^{1 + \alpha}}
            &=
            \sup_{t \in [0,T]}\left\|u^N - u\right\|_{\mathcal{C}^{1 + \alpha}}
            \\
            &=
            \sup_{t \in [0,T]} e^{\rho(T- t)} e^{-\rho(T- t)} \left\|u^N - u\right\|_{\mathcal{C}^{1 + \alpha}}
            \\
            &\leq
            e^{\rho T} \sup_{t \in [0,T]} e^{-\rho(T- t)} \left\|u^N - u\right\|_{\mathcal{C}^{1 + \alpha}}
            \\
            &=
            e^{\rho T} \left\|u^N - u\right\|_{C_T \mathcal{C}^{1 + \alpha}}^{(\rho)}.
        \end{split}
        \label{eq:norm_bounded_by_rho_norm}
    \end{equation}
    
    Plugging
    \eqref{eq:norm_bounded_by_rho_norm}
    into
    \eqref{eq:u-uN_in_Linfinity_bounded_by_u-uN_in_Calpha1}
    
    \begin{equation}
    \begin{split}
        \| u^N(t) - u(t) \|_{L^\infty}
        &\leq
        c \| u^N(t) - u(t) \|_{\mathcal{C}^{\alpha + 1}}
        \\
        &\leq
        \sup_{t \in [0,T]} c \left\|u^N(t) - u(t)\right\|_{\mathcal{C}^{\alpha + 1}}
        \\
        &=
        c \left\|u^N - u\right\|_{C_T \mathcal{C}^{\alpha + 1}}
        \\
        &\leq
        c e^{\rho T} \left\|u^N - u\right\|_{C_T \mathcal{C}^{\alpha + 1}}^{(\rho)}.
    \end{split}
    \end{equation}
    
    And finally by
    \eqref{eq:u-uNb-bN}
    
    \begin{equation}
        \left\|u^N(t) - u(t)\right\|_{L^\infty}
        \leq
        c \cdot c(\rho) \cdot e^{\rho T} \left\|b^N - b\right\|_{C_T \mathcal{C}^{-\beta}}
    \end{equation}
    
    which proves
   \eqref{eq:uNu_bounded_by_bNb}.
   
   For
   \eqref{eq:graduNu_bounded_by_bNb}
   recall that if
   $f \in \mathcal{C}^{1 + \alpha}$
   then
   $\nabla f \in \mathcal{C}^{\alpha}$.
   Also, by Bernstein inequality
   \cite[Eqn. (9)]{issoglio_mckean_2021}
   \begin{equation}
    \left\|\nabla f\right\|_{\mathcal{\alpha}}
    \leq
    c \left\|f\right\|_{\mathcal{1+ \alpha}}.
   \end{equation}
   
  Using the equivalent norm
    \begin{equation}
        \| f \|_{C^{1 + \alpha}}
        \leq
        c
        \left(
        \sup_{x \in \mathbb{R}^d} | f(x) |
        + \sup_{x \in \mathbb{R}^d} | \nabla f(x) |
        + \sup_{x \neq y \in \mathbb{R}^d}
        \frac{|\nabla f (x) - \nabla f(y)|}{|x - y|^\alpha}
        \right)
    \end{equation}
    
    we can see that
    
    \begin{equation}
        \begin{split}
        \left\|\nabla u^N(t) - \nabla u(t)\right\|_{L^\infty}
        &\leq
        c \left\|u^N(t) - u(t)\right\|_{\mathcal{C}^{1+\alpha}}.
        \end{split}
    \end{equation}
    
    And usign the same bounds that we used above for
    $c \left\|u^N(t) - u(t)\right\|_{\mathcal{C}^{1+\alpha}}$
    this point follows.
    
\end{proof}

\begin{proposition}%{Bound for $ \|\psi (t,x) - \psi^N(t,x) \|$}

    \label{prop:bound_psi-psiN}
    Take
    $\rho > \rho_0$
    as in
    Proposition \ref{prop:diff_u_uN},
    $N \to \infty$,
    $\kappa_\rho$
    from
    Proposition \ref{prop:diff_uN_graduN},
    and
    $\beta \in (0, 1/2)$,
    then we have
\begin{equation}
    \sup_{(t,x) \in [0,T] \times \mathbb{R}}
    \|\psi (t,x) - \psi^N(t,x)\| \leq 2 \kappa_\rho \left\|b - b^N\right\|_{C_T\mathcal{C}^{-\beta}}
\end{equation}
\end{proposition}

\begin{proof}
    Recall the definition of
    $\psi, \phi \in C_T \mathcal{C}^{1} $
    
    \begin{equation}
        \phi(t, x) := x + u(t, x)
    \end{equation}
    
    \begin{equation}
        \psi(t, \cdot) = \phi^{-1} (t, \cdot).
    \end{equation}
    
    Note that
    
    \begin{equation}
        u(y) = \int_0^1 \nabla u(\alpha y) y d \alpha + u(0).
    \end{equation}
    
    From there we have

    \begin{equation}
	    u(t, y) - u(t, y') = \int_0^1 \nabla u (t, \alpha(y - y')) (y - y') d \alpha
    \end{equation}

    and therefore

    \begin{equation}
	    \|u(t, y) - u(t, y')\| 
	    \geq
	    \left(\int_0^1 \| \nabla u (t, \alpha(y - y')) \|^2 d \alpha \right)^{1/2} \|y - y'\|,
    \end{equation}

    Lemma \ref{lemma:bounds_gradients}
    we finally have

    \begin{equation}
	    \begin{split}
		    \|u(t, y) - u(t, y'\|)
		    &\leq
		    \left(\frac{1}{4} \int_0^1 d \alpha \right)^{1/2} \|y - y\|'
		    \\
		    \|u(t, y) - u(t, y')\|^2
		    &\leq
		    \frac{1}{4} \|y - y'\|^2
	    \end{split}
    \end{equation}

\section{Convergence rate of the solution to the regularised SDE and the original}
In this section we present a bound for
$ \mathbbm{E} [X^N - X] $
in terms of
$ \| b^N - b \|_{C_T \mathcal{C}^{-\beta}} $.

\begin{proposition}

	\ref{as:b_in_ctcb}
	\ref{as:bN_converges_in_ctcb}

	$ \alpha > \beta $
	there is a constant
	$ C_\alpha $

	\begin{equation}
	\label{eq:EXN-X}
	  \mathbbm{E} [ X^N - X ] \leq C_{\alpha} \|b^N - b\|_{C_T \mathcal{C}^{-\beta}},
	\end{equation}

  $ N \to \infty $.

      |X^N_t - X_t|
			&=
			| \psi^N(t, \phi^N(t, X^N_t)) - \psi(t, \phi(t, X_t)) |
			\\
			&=
			| \psi^N(t, Y_t) - \psi(t, Y_t) |,
    \end{aligned}
	\end{equation}
	then adding and subtracting, and using the triangle inequality we get
