\section{Bound for the local time at zero of the solution to the SDEs}

We need a bound for
$\mathbb{E} [L^0_T ( Y^N - Y ) ]$,
for Sobolev spaces, this is result
\cite[Proposition 5.4]{de_angelis_numerical_2020}
we present it here for the solutions to the SDE blonging to the appropriate Besov spaces.

\begin{proposition}
				{Bound for local time of solutions to the SDE}
				{bound_local_time_sde}
	Let
	$b \in C_T \mathcal{C}^{-\beta}$
	and
	$b^N \to b$
	in
	$C_T \mathcal{C}^{-\beta}$
	as
	$N \to \infty$
	for
	$\beta \in (0, \frac{1}{4})$
	and for any
	$\alpha > \beta$


\end{proposition}

\begin{proof}
	Recall that
	$Y^N, Y$
	are solutions to the SDEs
	\begin{equation}
		Y_t = y_0 + \lambda \int_0^t u(s, \psi(s, Y_s)) ds + \int_0^t ( \nabla u (s, \psi(s, Y_t)) + 1 ) dW_s
	\end{equation}
	and
	\begin{equation}
		Y^N_t = y^N_0 + \lambda \int_0^t u^N(s, \psi^N(s, Y^N_s)) ds + \int_0^t ( \nabla u^N (s, \psi^N(s, Y^N_t)) + 1 ) dW_s
	\end{equation}
	so that the difference
	$Y^N - Y$
	is
	\begin{equation}
		\begin{split}
			Y^N - Y_t &= 
		(y^N_0 + \lambda \int_0^t u^N(s, \psi^N(s, Y^N_s)) ds + \int_0^t ( \nabla u^N (s, \psi^N(s, Y^N_t)) + 1 ) dW_s)
		\\
				  &-
		(y_0 + \lambda \int_0^t u(s, \psi(s, Y_s)) ds + \int_0^t ( \nabla u (s, \psi(s, Y_t)) + 1 ) dW_s)
		\\
				  &=
		(y^N_0 - y_0) + \lambda \int_0^t ( u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y_s)) ) ds
		\\
				  &-
		\int_0^t ( \nabla u^N (s, \psi^N(s, Y^N_t)) - \nabla u (s, \psi(s, Y_t))) dW_s,
		\end{split}
	\end{equation}

	and using
	\zref{lemma:local-time-at-0}
	we have the following bound
	
	\begin{equation}
		\begin{split}
			&\mathbb{E} [L_t^0 (Y^N - Y)]
			\\
			&\leq
			4 \epsilon +1
			\\
			&-
			2(1+\lambda)
			\mathbb{E}
			\qty[
				\int_0^t
				\qty( 
				\mathbbm{1}_{\qty{Y^N_s - Y_s \in (0, \epsilon)}} 
				+ \mathbbm{1}_\qty{Y^N_s - Y_s \geq \epsilon} e^{1 - \frac{Y^N_s - Y_s}{\epsilon}}
					)
			\qty( u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y_s))) ds
			]
		      	\\
			&+
			\frac{1}{\epsilon}
			\mathbb{E}
			\qty[
			\int_0^t \mathbbm{1}_\qty{Y^N_s - Y_s > \epsilon} e^{1 - \frac{Y^N_s - Y_s}{\epsilon}}
			\qty( \nabla u^N(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s)))^2 ds
			].
		\end{split}
		\label{eq:local_time_diff}
	\end{equation}

	\todo{add the explanation of why to drop the difussion term}

	For the second and third terms of
	\zref{eq:local_time_diff}
	let us bound the factors involving the differences of
	$u, u^N$
	and
	$\nabla u, \nabla u^N$.

	First, for $u, u^N$ adding and subtracting terms and using triangle inequality we have

	\begin{equation}
		\begin{split}
			\abs{u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y_s))}
			&\leq
			\abs{u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi^N(s, Y^N_s))}
			\\
			&+
			\abs{u(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y^N_s))}
			\\
			&+
			\abs{u(s, \psi(s, Y^N_s)) - u(s, \psi(s, Y_s))}.
		\end{split}
	\end{equation}
\end{proof}

The terms in the right hand side will be bounded as follows:

\begin{itemize}
\item
For the first term, by
\zref{prop:diff_uN_graduN}

\begin{equation}
  \abs{u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi^N(s, Y^N_s))}
	\leq
	\norm{ u^N(s) - u(s) }_{L^\infty}
	\leq
	\kappa_\rho \norm{ b - b^N }_{C_T\mathcal{C}^{-\beta}},
\end{equation}

\item
for the second term, observe that
$u, u^N$
are
$\frac{1}{2}$-Lipschitz
and by
\zref{prop:bound_psi-psiN}
we get

\begin{equation}
  \begin{split}
    \abs{u(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y^N_s))}
    &\leq
    \frac{1}{2}\abs{\psi^N(s, Y^N_s) - \psi(s, Y^N_s)}
    \\
    &\leq
    \kappa_\rho \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}},
  \end{split} 
\end{equation}

\item
and for the final term, note that
$\psi, \psi^N$
are
2-Lipschitz

so that

\begin{equation}
  \abs{u(s, \psi(s, Y^N_s)) - u(s, \psi(s, Y_s))}
	\leq
	\frac{1}{2} \abs{\psi(s, Y^N_s) - \psi(s, Y_s)}
	\leq
	\abs{Y^N_s - Y_s}.
\end{equation}
\end{itemize}

So that the following bound holds

\begin{equation}
	\abs{u^N(s, \psi^N(s, Y^N_s)) - u(s, \psi(s, Y_s))}
	\leq
  2\kappa_\rho \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}},
  +
	\abs{Y^N_s - Y^N}.
	\label{eq:bound_u_abs}
\end{equation}

Now for the third term in
\zref{eq:local_time_diff}
by adding and subtracting terms and using the triangle inequality

\begin{equation}
  \begin{split}
				\abs{\nabla u^N(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s))}
				&\leq
				\abs{\nabla u^N(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi^N(s, Y^N_s))}
				\\
				&+
				\abs{\nabla u(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi(s, Y^N_s))}
				\\
				&+
				\abs{\nabla u(s, \psi(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s))}
  \end{split}.
\end{equation}

The terms on the right hand side will be bounded as follows:

\begin{itemize}
\item
For the first term we use
\zref{prop:diff_uN_graduN}
and we have

\begin{equation}
  \begin{split}
    \abs{\nabla u^N(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi^N(s, Y^N_s))}
		&\leq
    \norm{\nabla u^N(s) - \nabla u(s) }_{L^\infty}
		\\
		&\leq
	  \kappa_\rho \norm{ b - b^N }_{C_T\mathcal{C}^{-\beta}},
  \end{split}
\end{equation}

for the second term see that
$\nabla u, \nabla u^N$
are
$\alpha$-HÃ¶lder
continuous and using
\zref{prop:bound_psi-psiN}
we have

\begin{equation}
\begin{split}
  \abs{\nabla u(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi(s, Y^N_s))}
	&\leq
	\abs{\psi^N(s, Y^N_s) - \psi(s, Y^N_s)}^\alpha \norm{u}_{C_T \mathcal{C}^{1 + \alpha}}
	\\
	&\leq
	(2 \kappa_\rho \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}})^\alpha \norm{u}_{C_T \mathcal{C}^{1 + \alpha}}.
\end{split}
\end{equation}

\end{itemize}

Therefore we get the bound

\begin{equation}
  \begin{split}
				\abs{\nabla u^N(s, \psi^N(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s))}
				&\leq
	  \kappa_\rho \norm{ b - b^N }_{C_T\mathcal{C}^{-\beta}}
				\\
				&+
	2^\alpha \kappa_\rho^\alpha \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}}^\alpha \norm{u}_{C_T \mathcal{C}^{1 + \alpha}}
				\\
				&+
				\abs{\nabla u(s, \psi(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s))}
  \end{split}.
	\label{eq:bound_gradu_abs}
\end{equation}

Using the bounds for
\zref{eq:bound_u_abs}
and
\zref{eq:bound_gradu_abs}
and the inequality

\begin{equation}
				(x_1 + \dotsm + x_k)^2 \leq k(x_1 + \dotsm + x_k),
\end{equation}

for some
$k$,
we get

\begin{equation}
\begin{split}
&\mathbb{E} [L^0_t ( Y^N - Y)]
\\
&\leq
4 \epsilon
+
4 (1 + \lambda) \qty(
  2\kappa_\rho \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}} t
  +
	\mathbb{E}\qty[\int_0^t\abs{Y^N_s - Y^N} ds]
	)
	\\
&+
\frac{1}{\epsilon} 3 t \norm{ b^N - b}_{C_T \mathcal{C}^{-\beta}}
\qty(
\kappa_\rho^2 \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}}
+
4(2 \kappa_\rho)^{2 \alpha} \norm{u}_{C_T \mathcal{C}^{1 + \alpha}}^2 \norm{b^N - b}_{C_T \mathcal{C}^{-\beta}}^{2 \alpha - 1}
)
\\
&+
\frac{1}{\epsilon} 3 \mathbb{E}
\qty(
			\int_0^t \mathbbm{1}_\qty{Y^N_s - Y_s > \epsilon} e^{1 - \frac{Y^N_s - Y_s}{\epsilon}} \qty| \nabla u(s, \psi(s, Y^N_s)) - \nabla u(s, \psi(s, Y_s))|^2 ds
)
\end{split}
\end{equation}
